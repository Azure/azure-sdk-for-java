                .withPlatform("dotnetcore")
                .withPlatformVersion("7.0")
                .withEnvironmentVariables(Arrays.asList(new EnvironmentVariable().withName("foo1").withValue("bar1"),

                .withPreBuildSteps(Arrays.asList(
